first_state = safe
death_state = destroyed

script{
    safe {
        on_enter {
            timer 3
            animation safe
        }
        on_timer {
            state starting
        }
    }
    starting {
        on_enter {
            animation starting
        }
        on_animation_end {
            state shooting
        }
    }
    shooting {
        on_enter {
            timer 3
            animation shooting
        }
        on_timer {
            state stopping
        }
    }
    stopping {
        on_enter {
            animation stopping
        }
        on_animation_end {
            state safe
        }
    }
    destroyed {
        on_enter {
            animation destroyed
            special_function die_start
        }
    }
}

first_state = sprout
death_state = destroyed

init {
    set_var red false
    set_var yellow false
    set_var blue false
    set_var cur_color none
    set_var bloom_time 240
}

script {
    sprout {
        on_enter {
            set_animation sprout
            set_timer 1
        }
        on_timer {
            set_timer 1
            calculate bloom_time bloom_time - 1
            if var bloom_time <= 120
                set_state sapling
            end_if
        }
    }
    sapling {
        on_enter {
            set_animation sapling
            start_particles small_sparks 15 0 150
            set_timer 1
        }
        on_timer {
            set_timer 1
            calculate bloom_time bloom_time - 1
            if var bloom_time <= 0
                set_state blooming
            end_if
        }
    }
    blooming {
        on_enter {
            set_animation blooming
            start_particles small_sparks 15 0 150
        }
        on_animation_end {
            set_var cur_color blue
            set_state picking_next_color
        }
    }
    picking_next_color {
        on_enter {
            set_timer 3
            
            if var red != true
                if var yellow != true
                    if var blue != true
                        set_var red true
                    end_if
                end_if
            end_if
            
            if var cur_color = red
                set_var cur_color yellow
                set_state idling
            end_if
            
            if var cur_color = yellow
                set_var cur_color blue
                set_state idling
            end_if
            
            if var cur_color = blue
                set_var cur_color red
                set_state idling
            end_if
        }
    }
    idling {
        on_enter {
            if var cur_color = red
                if var red != true
                    set_state picking_next_color
                end_if
                set_animation idling_red no_restart
            end_if
            
            if var cur_color = yellow
                if var yellow != true
                    set_state picking_next_color
                end_if
                set_animation idling_yellow no_restart
            end_if
            
            if var cur_color = blue
                if var blue != true
                    set_state picking_next_color
                end_if
                set_animation idling_blue no_restart
            end_if
        }
        on_timer {
            set_state picking_next_color
        }
    }
    destroyed {
        on_enter {
            set_animation dying
            start_particles pellet_posy_10_petals 100 0 150
            start_dying
            if var cur_color = red
                spawn red
            end_if
            if var cur_color = yellow
                spawn yellow
            end_if
            if var cur_color = blue
                spawn blue
            end_if
        }
        on_animation_end {
            delete
        }
    }
}

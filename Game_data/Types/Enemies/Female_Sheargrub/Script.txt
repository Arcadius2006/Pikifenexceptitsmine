first_state = hidden
death_state = dying

script {
    hidden {
        on_enter {
            animation hidden
            hide true
            near_reach chase
            far_reach chase
        }
        on_opponent_in_reach {
            state emerging
        }
        on_leave {
            hide false
        }
    }
    
    emerging {
        on_enter {
            animation emerging
        }
        on_animation_end {
            state examining
        }
    }
    
    examining {
        on_enter {
            animation examining
        }
        on_opponent_in_reach {
            focus mob
            state chasing
        }
        on_animation_end {
            state going_home
        }
        on_pikmin_land {
            state crushed
        }
    }
    
    burying {
        on_enter {
            animation burying
        }
        on_animation_end {
            state hidden
        }
    }
    
    chasing {
        on_enter {
            animation moving
            move focused_mob
        }
        on_focus_off_reach {
            state going_home
        }
        on_far_from_home {
            state going_home
        }
        on_pikmin_land {
            state crushed
        }
    }
    
    going_home {
        on_enter {
            animation moving
            move home
        }
        on_reach_destination {
            state burying
        }
        on_pikmin_land {
            state crushed
        }
    }
    
    crushed {
        on_enter {
            state dying
        }
    }
    
    dying {
        on_enter {
            special_function die_start
            animation dying
        }
        on_animation_end {
            special_function die_end
        }
    }
    
}

first_state = resting
death_state = dying

script {
    resting_start {
        on_enter {
            animation resting
            move stop
            timer 1
            gravity 1
        }
        on_timer {
            state resting
        }
        on_pikmin_land {
            state dying
        }
    }
    
    going_home {
        on_enter {
            move home
            timer 5
        }
        on_reach_destination {
            state resting_start
        }
        on_pikmin_land {
            state dying
        }
        on_timer {
            state flying
        }
    }
    
    resting {
        on_enter {
            animation resting
            timer 5
            near_reach fear
            far_reach fear
        }
        on_timer {
            state flying
        }
        on_opponent_in_reach {
            state flying
        }
        on_touch_opponent {
            state flying
        }
        on_pikmin_land {
            state dying
        }
    }
    
    flying {
        on_enter {
            animation flapping
            gravity -2
            timer 0.3
            move randomly
            var flutters 0
        }
        on_pikmin_land {
            state dying
        }
        on_timer {
            timer 0.3
            move stop vertically
            gravity 0
            move randomly
            inc_var flutters
            if var flutters = 10
                state going_home
            endif
        }
    }
    
    dying {
        on_enter {
            special_function die_start
            animation dying
        }
        on_animation_end {
            special_function die_end
        }
    }
    
}

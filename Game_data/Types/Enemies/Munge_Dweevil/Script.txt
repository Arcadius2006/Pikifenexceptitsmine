first_state = relaxed

script{
    relaxed{
        on_enter{
            animation = idling
            move = stop
        }
        on_see_opponent{
            focus = opponent
            state = panicked
        }
        on_big_damage{
            state = shaking
        }
        on_death{
            state = dying
        }
    }
    
    panicked{
        on_enter{
            animation = walking
            timer = 0.3
            move = randomly
        }
        on_timer{
            timer = 0.3
            move = randomly
        }
        on_lose_focused_mob{
            state = relaxed
        }
        on_far_from_home{
            state = going_home
        }
        on_big_damage{
            state = shaking
        }
        on_bottomless_pit{
            state = deleting
        }
        on_death{
            state = dying
        }
    }
    
    going_home{
        on_enter{
            move = home
        }
        on_reach_destination{
            state = relaxed
        }
        on_big_damage{
            state = shaking
        }
        on_bottomless_pit{
            state = deleting
        }
        on_death{
            state = dying
        }
    }
    
    shaking{
        on_enter{
            move = stop
            animation = shaking
        }
        on_animation_end{
            state = attacking
        }
    }
    
    attacking{
        on_enter{
            animation = attacking
            particle = gas_explosion
        }
        on_animation_end{
            state = relaxed
            particle = 
        }
    }
    
    dying{
        on_enter{
            special_function = die_start
            animation = dying
            move = stop
            gravity = 1
        }
        on_animation_end{
            special_function = die_end
        }
    }
    
    deleting{
        on_enter{
            special_function = die_end
            special_function = delete
        }
    }
    
}

first_state = examining
death_state = dying

init {
    var ignoring false
}

script {
    examining {
        on_enter {
            animation looking
            move stop
            near_reach search
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                state chasing
            endif
        }
        on_itch {
            state shaking
        }
        on_pikmin_land {
            state crushed
        }
        on_animation_end {
            state grazing
        }
    }
    
    grazing {
        on_enter {
            animation grazing
            move stop
        }
        on_pikmin_land {
            state crushed
        }
        on_animation_end {
            state exploring
        }
    }
    
    exploring {
        on_enter {
            move randomly
            animation moving
            timer 2
        }
        on_opponent_in_reach {
            focus mob
            state chasing
        }
        on_reach_destination {
            state examining
        }
        on_timer {
            state examining
        }
        on_far_from_home {
            state returning
        }
        on_pikmin_land {
            state crushed
        }
    }
    
    chasing {
        on_enter {
            animation moving
            move focused_mob
            near_reach attack
            far_reach chase
        }
        on_opponent_in_reach {
            state attacking
        }
        on_focus_off_reach {
            state examining
        }
        on_far_from_home {
            var ignoring true
            state returning
        }
        on_itch {
            state shaking
        }
        on_pikmin_land {
            state crushed
        }
    }
    
    attacking {
        on_enter {
            move stop
            animation attacking
            chomp 1 mouth
        }
        on_animation_end {
            state checking_mouth
        }
    }
    
    checking_mouth {
        on_leave {
            chomp =
        }
        on_mouth_occupied {
            state eating
        }
        on_mouth_empty {
            state examining
        }
        on_pikmin_land {
            state crushed
        }
    }
    
    eating {
        on_enter {
            animation eating
        }
        on_animation_end {
            eat all
            state examining
        }
        on_pikmin_land {
            state crushed
        }
    }
    
    shaking {
        on_enter {
            move stop
            animation shaking
        }
        on_animation_end {
            state examining
        }
        on_pikmin_land {
            state crushed
        }
    }
    
    returning {
        on_enter {
            move home
            animation moving
            timer 2
            near_reach search
        }
        on_leave {
            timer 0
        }
        on_timer {
            var ignoring false
        }
        on_reach_destination {
            state examining
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                state chasing
            endif
        }
    }
    
    crushed {
        on_enter {
            animation crushed
            health 0
            move stop
        }
        on_animation_end {
            state dying
        }
    }
    
    dying {
        on_enter {
            special_function die_start
            animation dying
        }
        on_animation_end {
            special_function die_end
        }
    }
    
}

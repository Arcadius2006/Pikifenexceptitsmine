first_state = idling
death_state = dying

script {
    idling {
        on_enter {
            var ignoring false
            animation idling
            move stop
            near_reach search
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                state chasing
            endif
        }
        on_itch {
            state shaking
        }
    }
    
    chasing {
        on_enter {
            animation moving
            move focused_mob
            near_reach attack
            far_reach chase
        }
        on_opponent_in_reach {
            state attacking
        }
        on_focus_off_reach {
            state returning
        }
        on_far_from_home {
            var ignoring true
            state returning
        }
        on_itch {
            state shaking
        }
    }
    
    attacking {
        on_enter {
            move stop
            animation attacking
            chomp 2 mouth
        }
        on_animation_end {
            state checking_mouth
        }
    }
    
    checking_mouth {
        on_mouth_occupied {
            state eating
        }
        on_animation_end {
            state idling
        }
    }
    
    eating {
        on_enter {
            chomp =
            animation eating
        }
        on_animation_end {
            eat all
            state idling
        }
    }
    
    shaking {
        on_enter {
            move stop
            animation shaking
        }
        on_animation_end {
            state idling
        }
    }
    
    returning {
        on_enter {
            move home
            animation moving
            timer 2
            near_reach search
        }
        on_leave {
            timer 0
        }
        on_timer {
            var ignoring false
        }
        on_reach_destination {
            state idling
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                state chasing
            endif
        }
    }
    
    dying {
        on_enter {
            special_function die_start
            animation dying
        }
        on_animation_end {
            special_function die_end
        }
    }
    
}

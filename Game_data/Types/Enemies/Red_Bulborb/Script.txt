first_state = sleeping
death_state = dying

script {
    sleeping {
        on_enter {
            var ignoring false
            animation sleeping
            near_reach wake_up
        }
        on_opponent_in_reach {
            state waking_up
        }
        on_damage {
            state waking_up
        }
        on_touch_opponent {
            state waking_up
        }
    }
    
    waking_up {
        on_enter {
            animation waking_up
        }
        on_animation_end {
            state examining
        }
    }
    
    examining {
        on_enter {
            animation looking
            move stop
            near_reach search
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                state turning
            endif
        }
        on_itch {
            state shaking
        }
        on_animation_end {
            state returning
        }
    }
    
    turning {
        on_enter {
            animation moving
            move stop
            face focused_mob
            near_reach chase
            far_reach close
        }
        on_tick {
            face focused_mob
        }
        on_opponent_in_reach {
            focus mob
            state chasing
        }
        on_focus_off_reach {
            state examining
        }
        on_itch {
            state shaking
        }
    }
    
    chasing {
        on_enter {
            animation moving
            move focused_mob
            near_reach attack
            far_reach chase
        }
        on_opponent_in_reach {
            state attacking
        }
        on_focus_off_reach {
            state examining
        }
        on_far_from_home {
            var ignoring true
            state examining
        }
        on_itch {
            state shaking
        }
    }
    
    attacking {
        on_enter {
            move stop
            animation attacking
            chomp 5 mouth
        }
        on_animation_end {
            state checking_mouth
        }
    }
    
    checking_mouth {
        on_mouth_occupied {
            state eating
        }
        on_mouth_empty {
            state failing
        }
    }
    
    eating {
        on_enter {
            chomp =
            animation eating
        }
        on_animation_end {
            eat all
            state examining
        }
    }
    
    failing {
        on_enter {
            animation failing
        }
        on_animation_end {
            state examining
        }
    }
    
    shaking {
        on_enter {
            move stop
            animation shaking
        }
        on_animation_end {
            state examining
        }
    }
    
    returning {
        on_enter {
            move home
            animation moving
            timer 2
            near_reach search
        }
        on_leave {
            timer 0
        }
        on_timer {
            var ignoring false
        }
        on_reach_destination {
            state falling_asleep
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                state turning
            endif
        }
    }
    
    falling_asleep {
        on_enter {
            animation falling_asleep
        }
        on_animation_end {
            state sleeping
        }
    }
    
    dying {
        on_enter {
            special_function die_start
            animation dying
        }
        on_animation_end {
            special_function die_end
        }
    }
    
}

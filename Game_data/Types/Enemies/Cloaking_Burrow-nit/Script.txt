first_state = hidden
death_state = dying

script {
    hidden {
        on_enter {
            set_var ignoring false
            set_near_reach wake_up
            set_animation hidden
        }
        on_opponent_in_reach {
            set_state emerging
        }
    }
    
    emerging {
        on_enter {
            set_animation emerging
        }
        on_animation_end {
            set_state examining
        }
    }
    
    examining {
        on_enter {
            set_animation looking
            stop
            set_near_reach search
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                set_state chasing
            end_if
        }
        on_itch {
            set_state shaking
        }
        on_animation_end {
            set_state returning
        }
    }
    
    chasing {
        on_enter {
            set_animation moving
            move focused_mob
            set_near_reach attack
            set_far_reach chase
        }
        on_opponent_in_reach {
            set_state attacking
        }
        on_focus_off_reach {
            set_state examining
        }
        on_far_from_home {
            set_var ignoring true
            set_state examining
        }
        on_itch {
            set_state shaking
        }
    }
    
    attacking {
        on_enter {
            stop
            set_animation attacking
            set_chomp_body_parts 1 mouth
        }
        on_animation_end {
            set_state checking_mouth
        }
    }
    
    checking_mouth {
        on_mouth_occupied {
            set_state eating
        }
        on_mouth_empty {
            set_state examining
        }
    }
    
    eating {
        on_enter {
            set_chomp_body_parts =
            set_animation eating
        }
        on_animation_end {
            eat all
            set_state examining
        }
    }
    
    shaking {
        on_enter {
            stop
            set_animation shaking
        }
        on_animation_end {
            set_state examining
        }
    }
    
    returning {
        on_enter {
            move home
            set_animation moving
            set_timer 2
            set_near_reach search
        }
        on_leave {
            set_timer 0
        }
        on_timer {
            set_var ignoring false
        }
        on_reach_destination {
            set_state hiding
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                set_state chasing
            end_if
        }
    }
    
    hiding {
        on_enter {
            set_animation hiding
        }
        on_animation_end {
            set_state hidden
        }
    }
    
    dying {
        on_enter {
            start_dying
            set_animation dying
        }
        on_animation_end {
            finish_dying
        }
    }
    
}

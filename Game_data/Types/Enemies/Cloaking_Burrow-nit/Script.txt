first_state = hidden
death_state = dying

script {
    hidden {
        on_enter {
            var ignoring false
            near_reach wake_up
            animation hidden
        }
        on_opponent_in_reach {
            state emerging
        }
    }
    
    emerging {
        on_enter {
            animation emerging
        }
        on_animation_end {
            state examining
        }
    }
    
    examining {
        on_enter {
            animation looking
            move stop
            near_reach search
        }
        on_opponent_in_reach {
            if ignoring false
                focus mob
                state chasing
            endif
        }
        on_itch {
            state shaking
        }
        on_animation_end {
            state returning
        }
    }
    
    chasing {
        on_enter {
            animation moving
            move focused_mob
            near_reach attack
            far_reach chase
        }
        on_opponent_in_reach {
            state attacking
        }
        on_focus_off_reach {
            state examining
        }
        on_far_from_home {
            var ignoring true
            state examining
        }
        on_itch {
            state shaking
        }
    }
    
    attacking {
        on_enter {
            move stop
            animation attacking
            chomp 1 mouth
        }
        on_animation_end {
            state checking_mouth
        }
    }
    
    checking_mouth {
        on_mouth_occupied {
            state eating
        }
        on_mouth_empty {
            state examining
        }
    }
    
    eating {
        on_enter {
            chomp =
            animation eating
        }
        on_animation_end {
            eat all
            state examining
        }
    }
    
    shaking {
        on_enter {
            move stop
            animation shaking
        }
        on_animation_end {
            state examining
        }
    }
    
    returning {
        on_enter {
            move home
            animation moving
            timer 2
            near_reach search
        }
        on_leave {
            timer 0
        }
        on_timer {
            var ignoring false
        }
        on_reach_destination {
            state hiding
        }
        on_opponent_in_reach {
            if ignoring false
                focus mob
                state chasing
            endif
        }
    }
    
    hiding {
        on_enter {
            animation hiding
        }
        on_animation_end {
            state hidden
        }
    }
    
    dying {
        on_enter {
            special_function die_start
            animation dying
        }
        on_animation_end {
            special_function die_end
        }
    }
    
}

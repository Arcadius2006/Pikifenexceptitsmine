first_state = sniffing
death_state = dying

init {
    var ignoring false
}

script {
    sniffing {
        on_enter {
            animation sniffing
            move stop
            near_reach search
        }
        on_opponent_in_reach {
            if ignoring false
                focus mob
                state chasing
            endif
        }
        on_itch {
            state shaking
        }
        on_animation_end {
            state exploring
        }
    }
    
    exploring {
        on_enter {
            move randomly
            animation walking
            timer 2
        }
        on_opponent_in_reach {
            focus mob
            state chasing
        }
        on_reach_destination {
            state sniffing
        }
        on_timer {
            state sniffing
        }
        on_far_from_home {
            state returning
        }
    }
    
    chasing {
        on_enter {
            animation walking
            move focused_mob
            near_reach attack
            far_reach chase
        }
        on_opponent_in_reach {
            state attacking
        }
        on_focus_off_reach {
            state sniffing
        }
        on_far_from_home {
            var ignoring true
            state returning
        }
        on_itch {
            state shaking
        }
    }
    
    attacking {
        on_enter {
            move stop
            animation attacking
            chomp 5 tongue1 tongue2
        }
        on_animation_end {
            state checking_mouth
        }
    }
    
    checking_mouth {
        on_leave {
            chomp =
        }
        on_mouth_occupied {
            state eating
        }
        on_mouth_empty {
            state sniffing
        }
    }
    
    eating {
        on_enter {
            animation chomping
        }
        on_animation_end {
            eat all
            state sniffing
        }
    }
    
    shaking {
        on_enter {
            move stop
            animation shaking
        }
        on_animation_end {
            state sniffing
        }
    }
    
    returning {
        on_enter {
            move home
            animation walking
            timer 2
            near_reach search
        }
        on_leave {
            timer 0
        }
        on_timer {
            var ignoring false
        }
        on_reach_destination {
            state sniffing
        }
        on_opponent_in_reach {
            if ignoring false
                focus mob
                state chasing
            endif
        }
    }
    
    dying {
        on_enter {
            special_function die_start
            animation dying
        }
        on_animation_end {
            special_function die_end
        }
    }
    
}

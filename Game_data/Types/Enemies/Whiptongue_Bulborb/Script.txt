first_state = sniffing
death_state = dying

init {
    set_var ignoring false
}

script {
    sniffing {
        on_enter {
            set_animation sniffing
            stop
            set_near_reach search
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                set_state turning
            end_if
        }
        on_itch {
            set_state shaking
        }
        on_animation_end {
            set_state exploring
        }
    }
    
    exploring {
        on_enter {
            move randomly
            set_animation walking
            set_timer 2
        }
        on_opponent_in_reach {
            focus mob
            set_state turning
        }
        on_reach_destination {
            set_state sniffing
        }
        on_timer {
            set_state sniffing
        }
        on_far_from_home {
            set_state returning
        }
    }
    
    turning {
        on_enter {
            set_animation walking
            stop
            turn focused_mob
            set_near_reach chase
            set_far_reach close
        }
        on_tick {
            turn focused_mob
        }
        on_opponent_in_reach {
            focus mob
            set_state chasing
        }
        on_focus_off_reach {
            set_state sniffing
        }
        on_itch {
            set_state shaking
        }
    }
    
    chasing {
        on_enter {
            set_animation walking
            move focused_mob
            set_near_reach attack
            set_far_reach chase
        }
        on_opponent_in_reach {
            set_state attacking
        }
        on_focus_off_reach {
            set_state sniffing
        }
        on_far_from_home {
            set_var ignoring true
            set_state returning
        }
        on_itch {
            set_state shaking
        }
    }
    
    attacking {
        on_enter {
            stop
            set_animation attacking
            set_chomp_body_parts 5 tongue1 tongue2
        }
        on_animation_end {
            set_state checking_mouth
        }
    }
    
    checking_mouth {
        on_leave {
            set_chomp_body_parts
        }
        on_mouth_occupied {
            set_state eating
        }
        on_mouth_empty {
            set_state sniffing
        }
    }
    
    eating {
        on_enter {
            set_animation chomping
        }
        on_animation_end {
            eat all
            set_state sniffing
        }
    }
    
    shaking {
        on_enter {
            stop
            set_animation shaking
        }
        on_animation_end {
            set_state sniffing
        }
    }
    
    returning {
        on_enter {
            move home
            set_animation walking
            set_timer 2
            set_near_reach search
        }
        on_leave {
            set_timer 0
        }
        on_timer {
            set_var ignoring false
        }
        on_reach_destination {
            set_state sniffing
        }
        on_opponent_in_reach {
            if var ignoring = false
                focus mob
                set_state turning
            end_if
        }
    }
    
    dying {
        on_enter {
            start_dying
            set_animation dying
        }
        on_animation_end {
            finish_dying
        }
    }
    
}
